const defaultConfig = { game_title:"Flying Plane", motivation_message:"DON'T GIVE UP 🔥" }; let gameState = { y:0, velocity:0, gravity:0.3, lift:-6, timeAlive:0, gameOver:false, gameStarted:false, level:1, score:0, obstacles:[], trails:[], touching:false, direction:-1, animationId:null, musicStarted:false }; const plane = document.getElementById('plane'); const startScreen = document.getElementById('startScreen'); const gameOverScreen = document.getElementById('gameOverScreen'); const levelDisplay = document.getElementById('levelDisplay'); const scoreDisplay = document.getElementById('scoreDisplay'); const finalScore = document.getElementById('finalScore'); const finalLevel = document.getElementById('finalLevel'); const gameContainer = document.getElementById('gameContainer'); const startBtn = document.getElementById('startBtn'); const restartBtn = document.getElementById('restartBtn'); const btnUp = document.getElementById('btnUp'); const btnDown = document.getElementById('btnDown'); const bgMusic = document.getElementById("bgMusic"); function playMusic(){ if(!gameState.musicStarted){ bgMusic.play().catch(()=>{}); gameState.musicStarted=true; }} function stopMusic(){ bgMusic.pause(); bgMusic.currentTime=0; gameState.musicStarted=false; } function initGame(){ gameState.obstacles.forEach(o=>o.element.remove()); gameState.trails.forEach(t=>t.remove()); gameState = {...gameState, y:window.innerHeight/2-25, velocity:0, gravity:0.5, lift:-6, timeAlive:0, gameOver:false, gameStarted:true, level:1, score:0, obstacles:[], trails:[], touching:false, direction:-1}; plane.style.left='80px'; plane.style.top=gameState.y+'px'; plane.style.transform='rotate(0deg)'; startScreen.style.display='none'; gameOverScreen.style.display='none'; updateDisplay(); gameLoop(); } function updateDisplay(){ levelDisplay.textContent=`Level: ${gameState.level}`; scoreDisplay.textContent=`Score: ${gameState.score}`; } function createObstacle(){ const minGap=180,maxGap=280,gap=minGap+Math.random()*(maxGap-minGap),gapTop=Math.random()*(window.innerHeight-gap-100)+50; const topObs=document.createElement('div'); topObs.className='obstacle'; topObs.style.width='24px'; topObs.style.height=gapTop+'px'; topObs.style.top='0'; topObs.style.left=window.innerWidth+'px'; gameContainer.appendChild(topObs); const bottomObs=document.createElement('div'); bottomObs.className='obstacle'; bottomObs.style.width='24px'; bottomObs.style.height=(window.innerHeight-gapTop-gap)+'px'; bottomObs.style.bottom='0'; bottomObs.style.left=window.innerWidth+'px'; gameContainer.appendChild(bottomObs); gameState.obstacles.push({element:topObs,x:window.innerWidth,passed:false}); gameState.obstacles.push({element:bottomObs,x:window.innerWidth,passed:false}); } function createTrail(){ const trail=document.createElement('div'); trail.className='trail'; trail.style.left='70px'; trail.style.top=(gameState.y+20)+'px'; trail.style.opacity='0.6'; gameContainer.appendChild(trail); gameState.trails.push(trail); setTimeout(()=>{ trail.style.transition='opacity 0.5s'; trail.style.opacity='0'; setTimeout(()=>{ trail.remove(); const idx=gameState.trails.indexOf(trail); if(idx>-1) gameState.trails.splice(idx,1); },500); },100); } function moveObstacles(){ const speed=4+gameState.level*0.5; for(let i=gameState.obstacles.length-1;i>=0;i--){ const obs=gameState.obstacles[i]; obs.x-=speed; obs.element.style.left=obs.x+'px'; const pRect=plane.getBoundingClientRect(), oRect=obs.element.getBoundingClientRect(); if(!(pRect.rightoRect.right-5 || pRect.bottomoRect.bottom-5)){ endGame(); return;} if(!obs.passed && obs.x+24<80){ obs.passed=true; gameState.score+=5; } if(obs.x<-30){ obs.element.remove(); gameState.obstacles.splice(i,1); } } } function endGame(){ gameState.gameOver=true; gameState.gameStarted=false; if(gameState.animationId) cancelAnimationFrame(gameState.animationId); stopMusic(); finalScore.textContent=gameState.score; finalLevel.textContent=gameState.level; gameOverScreen.style.display='flex'; } function gameLoop(){ if(gameState.gameOver) return; if(gameState.touching) gameState.velocity=gameState.direction===-1?gameState.lift:-gameState.lift/2; gameState.velocity+=gameState.gravity; gameState.velocity=Math.min(gameState.velocity,10); gameState.y+=gameState.velocity; if(gameState.y<0){ gameState.y=0; gameState.velocity=0; } if(gameState.y>window.innerHeight-50){ endGame(); return; } plane.style.transform=`rotate(${Math.min(Math.max(gameState.velocity*4,-35),35)}deg)`; plane.style.top=gameState.y+'px'; gameState.timeAlive++; if(gameState.timeAlive%20===0) gameState.score++; if(gameState.score>gameState.level*100){ gameState.level++; gameState.gravity+=0.02; } if(gameState.timeAlive%120===0) createObstacle(); if(gameState.timeAlive%5===0) createTrail(); moveObstacles(); updateDisplay(); gameState.animationId=requestAnimationFrame(gameLoop); } // Input handlers function handleInputStart(e){ if(!gameState.gameStarted) return; gameState.touching=true; gameState.direction=-1; playMusic(); } function handleInputEnd(e){ gameState.touching=false; } // Event listeners gameContainer.addEventListener('touchstart',handleInputStart,{passive:false}); gameContainer.addEventListener('touchend',handleInputEnd,{passive:false}); gameContainer.addEventListener('mousedown',handleInputStart); gameContainer.addEventListener('mouseup',handleInputEnd); document.addEventListener('keydown',(e)=>{ if(e.code==='Space'||e.code==='ArrowUp'||e.code==='KeyR'){e.preventDefault(); handleInputStart(e);} }); document.addEventListener('keyup',(e)=>{ if(e.code==='Space'||e.code==='ArrowUp'||e.code==='KeyR'){e.preventDefault(); handleInputEnd(e);} }); startBtn.addEventListener('click',(e)=>{ e.stopPropagation(); initGame(); }); restartBtn.addEventListener('click',(e)=>{ e.stopPropagation(); initGame();alert("don't give up");}); // Mobile buttons btnUp.addEventListener('touchstart',(e)=>{ e.preventDefault(); gameState.touching=true; gameState.touching=true; gameState.direction=1; playMusic();},{passive:false}); btnDown.addEventListener('touchend',(e)=>{ e.preventDefault(); gameState.touching=false;},{passive:false});
